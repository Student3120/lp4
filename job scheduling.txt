# Job Sequencing with Deadlines using a Greedy Method

# Each job has an ID, a profit, and a deadline
class Job:
    def __init__(self, job_id, deadline, profit):
        self.job_id = job_id
        self.deadline = deadline
        self.profit = profit


def job_sequencing(jobs):
    # Step 1: Sort jobs in descending order of profit (highest profit first)
    jobs.sort(key=lambda x: x.profit, reverse=True)

    # Step 2: Find the maximum deadline to create time slots
    max_deadline = max(job.deadline for job in jobs)

    # Step 3: Create an array to store job order (initially empty)
    # -1 means the slot is free
    slots = [-1] * (max_deadline + 1)

    total_profit = 0

    # Step 4: Iterate through all jobs
    for job in jobs:
        # Try to schedule the job at the latest possible time before its deadline
        for day in range(job.deadline, 0, -1):
            if slots[day] == -1:  # If this time slot is free
                slots[day] = job.job_id  # Assign job to this day
                total_profit += job.profit  # Add profit
                break  # Move to the next job once scheduled

    # Step 5: Display the job order and total profit
    print("\nJob sequence to get maximum profit:")
    for i in range(1, len(slots)):
        if slots[i] != -1:
            print(f"Day {i}: Job {slots[i]}")

    print(f"\nTotal Profit: {total_profit}")


# Example usage
n = int(input("Enter number of jobs: "))
jobs = []

for i in range(n):
    job_id = input(f"\nEnter Job ID for job {i+1}: ")
    deadline = int(input("Enter deadline: "))
    profit = int(input("Enter profit: "))
    jobs.append(Job(job_id, deadline, profit))

job_sequencing(jobs)
